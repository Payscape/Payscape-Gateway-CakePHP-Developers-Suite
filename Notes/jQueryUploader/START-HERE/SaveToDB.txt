    public $model = 'Property';


    
            
            
    public function saveImage($property_id, $changed_name){
        if((isset($property_id))&&(isset($changed_name))){
 
            $this->$model->PropertyImage->set('name', $changed_name);
            $this->$model->PropertyImage->set('image', $changed_name);
            $this->$model->PropertyImage->set('property_id', $property_id);
            $this->$model->PropertyImage->save();
        }
    }
